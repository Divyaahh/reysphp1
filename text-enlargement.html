// Function to change text size function changeSize(action) { var textElement = document.getElementById('enlargeable-text'); // Get the text element // Get the current font size of the text element var currentSize = window.getComputedStyle(textElement, null).getPropertyValue('font-size'); currentSize = parseFloat(currentSize); // Convert it to a number // If the action is 'increase', increase the font size by 2px if (action === 'increase') { textElement.style.fontSize = (currentSize + 2) + 'px'; } // If the action is 'decrease', decrease the font size by 2px else if (action === 'decrease') { textElement.style.fontSize = (currentSize - 2) + 'px'; } }
